Apps Script — Código de ejemplo
Creá un nuevo proyecto en script.google.com, pegá este código, publicá como Web App con acceso "Cualquier persona".
function doPost(e) { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp .openById("TU_SPREADSHEET_ID"); var sheet = ss.getSheetByName("Movimientos") || ss.insertSheet("Movimientos"); if (sheet.getLastRow() === 0) { sheet.appendRow([ "Fecha","Hora","Lote","Producto", "Remito","Pallet","Cantidad","Ubicación" ]); } var now = new Date(); sheet.appendRow([ Utilities.formatDate(now, "America/Argentina/Buenos_Aires", "dd/MM/yyyy"), Utilities.formatDate(now, "America/Argentina/Buenos_Aires", "HH:mm:ss"), data.lote, data.producto, data.remito, data.pallet, data.cantidad, data.ubicacion ]); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); }
